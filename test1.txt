func void sierpinski (double length, int depth) {
  if (depth == 0) {
    int i;
    for(i = 0; i < 3; i = i + 1) {
      move(length);
      rotate(-120);
    }
  }
  else {
    sierpinski(length/2, depth - 1);
    move(length/2);
    sierpinski(length/2, depth - 1);
    move(length/-2);
    rotate(-60);
    move(length/2);
    rotate(60);
    sierpinski(length/2, depth - 1);
    rotate(-60);
    move(length/-2);
    rotate(60);
  }
}

main cheese() {
  pup();
  move(-50);
  pdown();
  sierpinski(300, 4);
}
