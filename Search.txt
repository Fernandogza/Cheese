//función que que regresa el indice de la
//primera aparición de un numero 'n'
//en los primeros k numeros de un arreglo
//regresa -1 si no se encuentra;
func int search(int arr[], int n, int k){
   int i;
   for(i = 0; i < k; i = i + 1) {
      if(arr[i] == n){
         return i;
      }
   }
   return -1;
}

main cheese() {
   int test [10];
   test[5] = 1;
   print(search(test, 1, 10));
}
